path :String|Object

创建链接元素时的属性
Type:
  • String | Object
Source:
  • path.js, line 5
Example
import digi from 'digi'
import router from 'digi-router'

digi.plugins([ ...router ])

digi([
  { to: '/',     text: '首页' },
  { to: '/list', text: '列表页' },
  {
    path: '/',
    child: { text: '首页内容' }
  },
  {
    path: '/list'
    child: [
      { to: '/list/1', text: '详情页1' },
      { to: '/list/2', text: '详情页2' }
    ]
  },
  {
    path: { pathname: '/list/:id', watch: ({ params }) => console.log(params) // => { id: 1 } },
    text: '详情页内容'
  },
  {
    path: /.+/,
    text: '404页面放在最后'
  }
])
